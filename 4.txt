create table Employe ( Eno number(3), Ename varchar2(20), Address varchar2(30),
Email varchar(20), Salary number );

create table Employe2 ( Eno number(3), Ename varchar2(20), Address  varchar2(30), Email varchar(20), Salary number ); 


CREATE DATABASE LINK "LINK1" USING 'XE';
CREATE DATABASE LINK "LINK2" USING 'XE'; 


create or replace Trigger insert_data  
after insert on Employe  
for each row  
begin  
insert into Employe2@LINK2  
values (:new.Eno,:new.Ename,:new.Address,:new.Email,:new.Salary);  end; 
/ 

create or replace Trigger del_data  
before delete on Employe  
for each row  
begin  
delete from Employe2@LINK2  
where Eno=:old.Eno;  
end; 
/ 

create or replace Trigger update_data  
after update on Employe  
for each row  
begin  
update Employe2@LINK2  
set Eno =:new.Eno,  
Ename =:new.Ename,  
Address = :new.Address,  
Email=:new.Email,  
Salary=:new.Salary 
where Eno=:old.Eno;  
end; 
/
