connect system/123

create public database link emp_link1 connect to scott identified by admin using 'XE';
create public database link emp_link2 connect to mscit identified by admin using 'XE';

select emp1.eno@emp_link1,name,address,email,salary from emp1@emp_link1,emp2@emp_link2 where emp1.eno@emp_link1 = emp2.eno@emp_link2;

select emp1.eno@emp_link1,name,salary from emp1@emp_link1,emp2@emp_link2 where emp1.eno@emp_link1 = emp2.eno@emp_link2 and emp2.eno@emp_link2 = 1;

select emp1.eno@emp_link1,emp2.email@emp_link2 from emp1@emp_link1,emp2@emp_link2 where emp1.name@emp_link1 = 'pavan' and emp1.eno@emp_link1 = emp2.eno@emp_link2;

select emp1.eno@emp_link1,name,email from emp1@emp_link1,emp2@emp_link2 where emp1.eno@emp_link1 = 2 and emp1.eno@emp_link1 = emp2.eno@emp_link2;

select emp1.name@emp_link1, emp2.salary@emp_link2 from emp1@emp_link1,emp2@emp_link2 where emp2.salary@emp_link2 >5000 and emp1.eno@emp_link1 = emp2.eno@emp_link2;
